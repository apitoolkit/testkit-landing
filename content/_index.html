describe('User API tests', () => {
    it('should get the current user', () => {
        cy.request({
            method: 'GET',
            url: Cypress.env('APIURL') + '/user',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': 'Bearer ' + Cypress.env('TOKEN')
            }
        }).then((response) => {
            expect(response.body.user).to.exist;
            expect(response.body.user.username).to.equal('Jon Doe');
            expect(response.body.user.bio).to.be.a('string').and.not.empty;
            expect(response.body.user.updatedAt).to.match(/\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}/);
            expect(response.body.user.id).to.be.a('number');
            Cypress.env('userID', response.body.user.id);
            Cypress.env('updatedAt', response.body.user.updatedAt);
        });
    });
    it('should update the user', () => {
        cy.request({
            method: 'PUT',
            url: Cypress.env('APIURL') + '/user',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': 'Bearer ' + Cypress.env('TOKEN')
            },
            body: {
                user: {
                    email: Cypress.env('EMAIL'),
                    bio: 'hello world'
                }
            }
        }).then((updateResponse) => {
            expect(updateResponse.status).to.equal(200);
            expect(updateResponse.body.user.email).to.exist;
            expect(updateResponse.body.user.id).to.equal(Cypress.env('userID'));
            expect(updateResponse.body.user.updatedAt).not.to.equal(Cypress.env('updatedAt'));
        });
    });
});
